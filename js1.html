const readline = require("readline"); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); rl.question("Enter marks of Subject 1: ", (m1) => { rl.question("Enter marks of Subject 2: ", (m2) => { rl.question("Enter marks of Subject 3: ", (m3) => { rl.question("Enter marks of Subject 4: ", (m4) => { rl.question("Enter marks of Subject 5: ", (m5) => { let s1 = Number(m1); let s2 = Number(m2); let s3 = Number(m3); let s4 = Number(m4); let s5 = Number(m5); let sum = s1 + s2 + s3 + s4 + s5; let avg = sum / 5; let grade; if (avg >= 75) { grade = "Distinction"; } else if (avg >= 60) { grade = "First Class"; } else if (avg >= 50) { grade = "Second Class"; } else if (avg >= 40) { grade = "Pass"; } else { grade = "Fail"; } console.log("\nTotal Marks =", sum); console.log("Average Marks =", avg); console.log("Grade =", grade); rl.close(); }); }); }); }); });
